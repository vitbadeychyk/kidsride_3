Імпорт / синхронізація з XML

Вкажи URL XML-фіду постачальника (формат YML/Prom) — система сама зчитає товари, фото, ціни і запише в базу. Існуючі товари оновляться за артикулом (vendorCode).

Остання синхронізація
Ще не запускалась
Що зчитується з кожного <item>
vendorCode → артикул (ключ оновлення)
name_ua / name → назва товару
vendor → бренд
price, quantity_in_stock, available
усі <image> → фото галерея
description_ua → опис, <param> → характеристики
Синхронізація...0%
⚠️ Якщо браузер блокує (CORS)
Більшість постачальників не дозволяють пряме читання XML з браузера. Скрипт автоматично пробує через публічний проксі (corsproxy.io). Якщо й це не працює — завантаж файл XML вручну кнопкою вище.

🗃️ SQL для Supabase

Скопіюйте та виконайте в Supabase → SQL Editor для зберігання налаштувань:

-- Виконайте в Supabase → SQL Editor
-- (кожна вкладка Налаштувань = окрема таблиця)

-- 1. Магазин
CREATE TABLE IF NOT EXISTS settings_store (id int PRIMARY KEY DEFAULT 1, shop_name text, tagline text, phone text, email text, hours text, address text, logo_url text, about text, instagram text, facebook text, tiktok text, youtube text, updated_at timestamptz DEFAULT now());
INSERT INTO settings_store (id) VALUES (1) ON CONFLICT (id) DO NOTHING;
ALTER TABLE settings_store ENABLE ROW LEVEL SECURITY;
CREATE POLICY "admin_store" ON settings_store USING (auth.role()='authenticated') WITH CHECK (auth.role()='authenticated');

-- 2. Замовлення
CREATE TABLE IF NOT EXISTS settings_orders (id int PRIMARY KEY DEFAULT 1, order_prefix text DEFAULT 'KR-', min_order numeric DEFAULT 0, auto_close_days int DEFAULT 0, confirm_text text, manager_notes text, pay_cod bool DEFAULT true, pay_card bool DEFAULT true, pay_bank bool DEFAULT false, pay_crypto bool DEFAULT false, updated_at timestamptz DEFAULT now());
INSERT INTO settings_orders (id) VALUES (1) ON CONFLICT (id) DO NOTHING;
ALTER TABLE settings_orders ENABLE ROW LEVEL SECURITY;
CREATE POLICY "admin_orders_cfg" ON settings_orders USING (auth.role()='authenticated') WITH CHECK (auth.role()='authenticated');

-- 3. Сповіщення
CREATE TABLE IF NOT EXISTS settings_notifications (id int PRIMARY KEY DEFAULT 1, email_orders text, email_reviews text, tg_token text, tg_chat text, tg_new_order bool DEFAULT false, tg_new_review bool DEFAULT false, tg_low_stock bool DEFAULT false, updated_at timestamptz DEFAULT now());
INSERT INTO settings_notifications (id) VALUES (1) ON CONFLICT (id) DO NOTHING;
ALTER TABLE settings_notifications ENABLE ROW LEVEL SECURITY;
CREATE POLICY "admin_notify" ON settings_notifications USING (auth.role()='authenticated') WITH CHECK (auth.role()='authenticated');

-- 4. Доставка
CREATE TABLE IF NOT EXISTS settings_delivery (id int PRIMARY KEY DEFAULT 1, np_api_key text, np_sender_name text, np_sender_phone text, np_city text, np_warehouse text, np_weight numeric DEFAULT 1, np_size text, cod_fee_pct numeric DEFAULT 2, cod_fee_min numeric DEFAULT 20, cod_include_in_total bool DEFAULT false, pickup_enabled bool DEFAULT false, pickup_address text, updated_at timestamptz DEFAULT now());
INSERT INTO settings_delivery (id) VALUES (1) ON CONFLICT (id) DO NOTHING;
ALTER TABLE settings_delivery ENABLE ROW LEVEL SECURITY;
CREATE POLICY "admin_delivery" ON settings_delivery USING (auth.role()='authenticated') WITH CHECK (auth.role()='authenticated');

-- 5. Вигляд
CREATE TABLE IF NOT EXISTS settings_appearance (id int PRIMARY KEY DEFAULT 1, color_primary text DEFAULT '#FF6B35', color_nav text DEFAULT '#1B2A4A', orders_per_page int DEFAULT 20, ui_compact bool DEFAULT false, ui_animations bool DEFAULT true, ui_sound bool DEFAULT false, updated_at timestamptz DEFAULT now());
INSERT INTO settings_appearance (id) VALUES (1) ON CONFLICT (id) DO NOTHING;
ALTER TABLE settings_appearance ENABLE ROW LEVEL SECURITY;
CREATE POLICY "admin_appearance" ON settings_appearance USING (auth.role()='authenticated') WITH CHECK (auth.role()='authenticated');

-- 6. SEO
CREATE TABLE IF NOT EXISTS settings_seo (id int PRIMARY KEY DEFAULT 1, seo_title text, seo_description text, og_image text, ga_id text, fb_pixel text, site_url text, updated_at timestamptz DEFAULT now());
INSERT INTO settings_seo (id) VALUES (1) ON CONFLICT (id) DO NOTHING;
ALTER TABLE settings_seo ENABLE ROW LEVEL SECURITY;
CREATE POLICY "admin_seo" ON settings_seo USING (auth.role()='authenticated') WITH CHECK (auth.role()='authenticated');
